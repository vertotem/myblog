    <script>
        let historyData = [];
        let filteredData = [];
        let isDataLoaded = false;

        // 加载数据文件 (现在改为加载JSON文件)
        async function loadData() {
            try {
                const response = await fetch('./data.json');
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                const jsonData = await response.json();
                return jsonData;
            } catch (error) {
                console.error('读取data.json文件失败:', error);
                const timeline = document.getElementById('timeline');
                timeline.innerHTML = `
                    <div class="no-results">
                        <i class="fas fa-exclamation-triangle"></i>
                        <div>无法读取data.json文件</div>
                        <div style="font-size: 0.9rem; margin-top: 10px; color: #666;">
                            请确保data.json文件与HTML文件在同一目录下，并格式正确
                        </div>
                    </div>
                `;
                return null;
            }
        }
        
        // 渲染时间轴
        function renderTimeline(data) {
            const timeline = document.getElementById('timeline');
            
            if (data.length === 0) {
                timeline.innerHTML = `
                    <div class="no-results">
                        <i class="fas fa-search"></i>
                        <div>没有找到匹配的历史事件</div>
                    </div>
                `;
                return;
            }

            timeline.innerHTML = data.map((item, index) => `
                <div class="timeline-item" style="animation-delay: ${index * 0.1}s">
                    <div class="timeline-marker"></div>
                    <div class="timeline-content">
                        <div class="year">
                            <span>${item.displayYear.replace(/公元前?(\d+)年(?:（(.+?)）)?/, '$1年').replace('前', '公元前')}</span>
                            ${item.reignYear ? `<span class="reign-year">${item.reignYear}</span>` : ''}
                        </div>
                        <div class="event">${item.event}</div>
                    </div>
                </div>
            `).join('');
        }

        // 高亮搜索结果
        function highlightText(text, searchTerm) {
            if (!searchTerm) return text;
            const regex = new RegExp(`(${searchTerm})`, 'gi');
            return text.replace(regex, '<span class="highlight">$1</span>');
        }

        // 显示加载状态
        function showLoading() {
            const timeline = document.getElementById('timeline');
            timeline.innerHTML = `
                <div class="no-results">
                    <i class="fas fa-spinner fa-spin"></i>
                    <div>正在加载历史数据...</div>
                </div>
            `;
        }

        // 年份定位功能
        function jumpToYear(targetYear) {
            if (!isDataLoaded) return;
            
            const year = parseInt(targetYear);
            if (isNaN(year)) return;

            const filtered = historyData.filter(item => Math.abs(item.year - year) <= 20);
            if (filtered.length > 0) {
                renderTimeline(filtered);
                
                // 滚动到第一个匹配项
                setTimeout(() => {
                    const firstItem = document.querySelector('.timeline-item');
                    if (firstItem) {
                        firstItem.scrollIntoView({ behavior: 'smooth', block: 'center' });
                        firstItem.querySelector('.timeline-content').style.borderLeft = '5px solid #ff6b6b';
                        setTimeout(() => {
                            firstItem.querySelector('.timeline-content').style.borderLeft = '5px solid #667eea';
                        }, 2000);
                    }
                }, 100);
            } else {
                renderTimeline([]);
            }
        }

        // 搜索功能
        function searchEvents(searchTerm) {
            if (!isDataLoaded) return;
            
            if (!searchTerm) {
                filteredData = historyData;
            } else {
                filteredData = historyData.filter(item => 
                    item.event.includes(searchTerm) || 
                    item.displayYear.includes(searchTerm) ||
                    item.reignYear.includes(searchTerm)
                ).map(item => ({
                    ...item,
                    event: highlightText(item.event, searchTerm),
                    displayYear: highlightText(item.displayYear, searchTerm),
                    reignYear: highlightText(item.reignYear, searchTerm)
                }));
            }
            renderTimeline(filteredData);
        }

        // 显示/隐藏清除按钮
        function toggleClearButton(input, clearBtn) {
            if (input.value) {
                clearBtn.classList.add('show');
            } else {
                clearBtn.classList.remove('show');
            }
        }

        // 初始化
        async function init() {
            showLoading();
            
            const jsonData = await loadData();
            if (!jsonData) return;
            
            // JSON文件已经包含了排序，所以无需再次排序
            historyData = jsonData;
            filteredData = historyData;
            isDataLoaded = true;
            renderTimeline(filteredData);

            const yearInput = document.getElementById('yearInput');
            const searchInput = document.getElementById('searchInput');
            const clearYear = document.getElementById('clearYear');
            const clearSearch = document.getElementById('clearSearch');

            // 年份输入事件
            yearInput.addEventListener('input', (e) => {
                toggleClearButton(yearInput, clearYear);
                if (e.target.value) {
                    jumpToYear(e.target.value);
                } else {
                    renderTimeline(historyData);
                }
            });

            // 搜索输入事件
            searchInput.addEventListener('input', (e) => {
                toggleClearButton(searchInput, clearSearch);
                searchEvents(e.target.value);
            });

            // 清除按钮事件
            clearYear.addEventListener('click', () => {
                yearInput.value = '';
                clearYear.classList.remove('show');
                renderTimeline(historyData);
            });

            clearSearch.addEventListener('click', () => {
                searchInput.value = '';
                clearSearch.classList.remove('show');
                searchEvents('');
            });

            // 回车键快速跳转
            yearInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') {
                    jumpToYear(e.target.value);
                }
            });
        }

        // 页面加载完成后初始化
        document.addEventListener('DOMContentLoaded', init);
    </script>
